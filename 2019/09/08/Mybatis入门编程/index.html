MyBatis，XML方式两种编程

2019-09-08

第一种方式：以增删改查学生信息表为例，属性有id，name，age，score

第一步：建立学生表student，加入若干条记录

第二步：在src->main->java下创建student包，进而创建User类，对student表进行封装，使之成为一个对象

package student;

public class User {
private Integer id;
private String name;
private Integer age;
private Double score;

public Integer getId() {
return id;
}

public void setId(Integer id) {
this.id = id;
}

public String getName() {
return name;
}

public void setName(String name) {
this.name = name;
}

public Integer getAge() {
return age;
}

public void setAge(Integer age) {
this.age = age;
}

public Double getScore() {
return score;
}

public void setScore(Double score) {
this.score = score;
}

@Override
public String toString() {
return "User{" +
"id=" + id +
", name='" + name + '\'' +
", age=" + age +
", score=" + score +
'}';
}
}

第三步：在idea中src，main目录下建立resources文件夹，右键选择Mark Directory as，选择Resources root，在该文件加下加入日志文件，进而在resources文件夹下创建主配置目录SqlConfigMapper.xml,代码如下

<?xml version="1.0" encoding="UTF-8" ?>
<!DOCTYPE configuration
PUBLIC "-//mybatis.org//DTD Config 3.0//EN"
"http://mybatis.org/dtd/mybatis-3-config.dtd">
<!--<transactionManager type="JDBC" />
    该标签用于指定事务管理器。type的取值有两个：
    JDBC：使用JDBC事务管理器
    MANAGED：使用第三方事务管理器，例如使用Spring的事务管理器

    <dataSource type="POOLED">
    该标签用于指定数据源类型。type的取值有三个：
    POOLED：使用连接池技术（默认使用的是Mybatis内置连接池）
    UNPOOLED：不使用连接池技术
    JNDI：Java Naming-Directory Interface，Java命名与目录接口，这是一个容器。
-->

<!--主配置文件，连接到数据库-->
<configuration>
<!--配置环境-->
<environments default="mysql">
<!--配置mysql的环境，default取名无所谓，但指定之后id必须也为该指定-->
<environment id="mysql">
<!--配置事务的类型，当前事务为JDBC-->
<transactionManager type="JDBC"/>
<!--配置数据源（即数据库连接池）-->
<dataSource type="POOLED">
<!--配置数据库相关的四个基本信息-->
<property name="driver" value="com.mysql.jdbc.Driver"/>
<property name="url" value="jdbc:mysql://localhost:3306/test"/>
<property name="username" value="root"/>
<property name="password" value="1018"/>
</dataSource>
</environment>
</environments>


<!--指定映射配置文件的位置，映射配置文件位置为每个dao独立的配置文件
针对IUserDao进行的配置-->
<mappers>
<mapper resource="student/UserMapper.xml"/>
</mappers>
</configuration>

第四步：在src->main->java下创建dao包，进而创建StudentInte接口，其中书写了增，删，改，查四个方法，代码如下：

package dao;

import student.User;

import java.util.List;

public interface StudentInte {
List<User> findIdStudent(int id);
List<User> findAll();
void insertStudent(User user);
void updateStudent(User user);
void deleteStudent(int id);

}

第五步：在resources文件夹下创建同名student包，在其中写入User类的映射文件UserMapper.xml，代码如下：

<?xml version="1.0" encoding="UTF-8" ?>
<!DOCTYPE mapper
PUBLIC "-//mybatis.org//DTD Mapper 3.0//EN"
"http://mybatis.org/dtd/mybatis-3-mapper.dtd">
<!--配置User类的映射，放在resources文件夹下同名的student(便于查找)包内-->
<mapper namespace="dao.StudentInte"><!--此处取相对路径-->


<!--namespace:名称空间，一定不能为空，默认取本文件的相对位置
parameterType：参数类型，取值为该SQL语句中传递的参数类型，如按id查询中，参数id
为整型，则parameterType取值为Integer，删除语句同样如此
插入语句，对象类型为User类对象，所以parameterType取值为student.User
修改语句，对象类型为User类对象，所以parameterType取值为student.User
删除语句，参数id为整型，则parameterType取值为Integer
取该类相对位置
resultType：返回类型
按id查询结果为User类对象，所以resultType取值为student.User，查询所有同样如此
因插入，修改，删除无返回对象，故不用设置
取该类相对位置
id：起唯一标识作用，使相应功能找到相应SQL语句-->

<!--根据id查询某个学生的信息-->
<select id="findIdStudent" parameterType="Integer"
resultType="student.User">
select * from student where id = #{id}
</select>

<!--查询所有学生信息-->
<select id="findAll" resultType="student.User">
select * from student
</select>

<!--插入学生信息-->
<insert id="insertStudent" parameterType="student.User">
insert into student (name,age,score) value(#{name},#{age},#{score})
</insert>

<!--修改某个学生的信息-->
<update id="updateStudent" parameterType="student.User">
update student set name=#{name},age=#{age},score=#{score} where id = #{id}
</update>

<!--按id删除某个学生-->
<delete id="deleteStudent" parameterType="Integer">
delete from student where id = #{id}
</delete>


</mapper>

第六步：在test文件夹下，创建测试用的ibatis3包，创建测试类MyBatisTest，代码如下：

package ibatis3;
//功能：将一个学生信息写入到student
//创建student表
//表名：student
//字段：id,name,age,score

import dao.StudentInte;
import org.apache.ibatis.io.Resources;
import org.apache.ibatis.session.SqlSession;
import org.apache.ibatis.session.SqlSessionFactory;
import org.apache.ibatis.session.SqlSessionFactoryBuilder;
import student.User;

import java.io.IOException;
import java.io.InputStream;
import java.util.List;

public class Student {
public static void main(String[] args) {
try {
User student =new User();
InputStream in = Resources.getResourceAsStream("SqlConfigMapper.xml");

SqlSessionFactoryBuilder ssfb = new SqlSessionFactoryBuilder();

SqlSessionFactory ssf = ssfb.build(in);

SqlSession ss = ssf.openSession();

StudentInte stu = ss.getMapper(StudentInte.class);
//插入信息

student.setName("哈哈");
student.setAge(23);
student.setScore(99.0);
stu.insertStudent(student);

//修改信息
student.setId(3);
student.setName("天天");
student.setAge(55);
student.setScore(45.4);
stu.updateStudent(student);

//删除信息
stu.deleteStudent(2);

//按id查询
List<User> listId =stu.findIdStudent(1);
System.out.println(listId);
System.out.println("---------------");

//查询所有
List<User> list = stu.findAll();

System.out.println(list);
ss.commit();//完成之后一定提交事务，否则数据库可能会刷新不出来

//关闭资源
in.close();
ss.close();

} catch (IOException e) {
e.printStackTrace();
}

}
}

2，事务完成后注意提交且关闭资源文件

3.注意及时配置约束文件
