MyBatis入门编程

Mybatis xml两种开发方式

第一种方式

建立接口但是不建立实现类，使用SqlSession创建dao接口的代理对象,以增删改查学生信息表为例，属性有id，name，age，score（贺与练）

第一步：建立学生表student，加入若干条记录

第二步：在src->main->java下创建student包，进而创建User类，对student表进行封装，使之成为一个对象

package student;

public class User {
private Integer id;
private String name;
private Integer age;
private Double score;

public Integer getId() {
return id;
}

public void setId(Integer id) {
this.id = id;
}

public String getName() {
return name;
}

public void setName(String name) {
this.name = name;
}

public Integer getAge() {
return age;
}

public void setAge(Integer age) {
this.age = age;
}

public Double getScore() {
return score;
}

public void setScore(Double score) {
this.score = score;
}

@Override
public String toString() {
return "User{" +
"id=" + id +
", name='" + name + '\'' +
", age=" + age +
", score=" + score +
'}';
}
}

第三步：在idea中src，main目录下建立resources文件夹，右键选择Mark Directory as，选择Resources root，在该文件加下加入日志文件，进而在resources文件夹下创建主配置目录SqlConfigMapper.xml,代码如下：

<?xml version="1.0" encoding="UTF-8" ?>
<!DOCTYPE configuration
PUBLIC "-//mybatis.org//DTD Config 3.0//EN"
"http://mybatis.org/dtd/mybatis-3-config.dtd">
<!--<transactionManager type="JDBC" />
    该标签用于指定事务管理器。type的取值有两个：
    JDBC：使用JDBC事务管理器
    MANAGED：使用第三方事务管理器，例如使用Spring的事务管理器

    <dataSource type="POOLED">
    该标签用于指定数据源类型。type的取值有三个：
    POOLED：使用连接池技术（默认使用的是Mybatis内置连接池）
    UNPOOLED：不使用连接池技术
    JNDI：Java Naming-Directory Interface，Java命名与目录接口，这是一个容器。
-->

<!--主配置文件，连接到数据库-->
<configuration>
<!--配置环境-->
<environments default="mysql">
<!--配置mysql的环境，default取名无所谓，但指定之后id必须也为该指定-->
<environment id="mysql">
<!--配置事务的类型，当前事务为JDBC-->
<transactionManager type="JDBC"/>
<!--配置数据源（即数据库连接池）-->
<dataSource type="POOLED">
<!--配置数据库相关的四个基本信息-->
<property name="driver" value="com.mysql.jdbc.Driver"/>
<property name="url" value="jdbc:mysql://localhost:3306/test"/>
<property name="username" value="root"/>
<property name="password" value="1018"/>
</dataSource>
</environment>
</environments>

​




第四步：在src->main->java下创建dao包，进而创建StudentInte接口，其中书写了增，删，改，查四个方法，代码如下：

package dao;

import student.User;

import java.util.List;

public interface StudentInte {
List<User> findIdStudent(int id);
List<User> findAll();
void insertStudent(User user);
void updateStudent(User user);
void deleteStudent(int id);

}

第五步：在resources文件夹下创建同名student包，在其中写入User类的映射文件UserMapper.xml，代码如下：

<?xml version="1.0" encoding="UTF-8" ?>
<!DOCTYPE mapper
PUBLIC "-//mybatis.org//DTD Mapper 3.0//EN"
"http://mybatis.org/dtd/mybatis-3-mapper.dtd">
<!--配置User类的映射，放在resources文件夹下同名的student(便于查找)包内-->
<mapper namespace="dao.StudentInte"><!--此处取相对路径-->

​

<!--根据id查询某个学生的信息-->
<select id="findIdStudent" parameterType="Integer"
resultType="student.User">
select * from student where id = #{id}
</select>

<!--查询所有学生信息-->
<select id="findAll" resultType="student.User">
select * from student
</select>

<!--插入学生信息-->
<insert id="insertStudent" parameterType="student.User">
insert into student (name,age,score) value(#{name},#{age},#{score})
</insert>

<!--修改某个学生的信息-->
<update id="updateStudent" parameterType="student.User">
update student set name=#{name},age=#{age},score=#{score} where id = #{id}
</update>

<!--按id删除某个学生-->
<delete id="deleteStudent" parameterType="Integer">
delete from student where id = #{id}
</delete>

第六步：在test文件夹下，创建测试用的ibatis3包，创建测试类MyBatisTest，代码如下：

package ibatis3;
//功能：将一个学生信息写入到student
//创建student表
//表名：student
//字段：id,name,age,score

import dao.StudentInte;
import org.apache.ibatis.io.Resources;
import org.apache.ibatis.session.SqlSession;
import org.apache.ibatis.session.SqlSessionFactory;
import org.apache.ibatis.session.SqlSessionFactoryBuilder;
import student.User;

import java.io.IOException;
import java.io.InputStream;
import java.util.List;

public class Student {
public static void main(String[] args) {
try {
User student =new User();
//1.读取配置文件
InputStream in = Resources.getResourceAsStream("SqlConfigMapper.xml");
//2.创建SqlSessionFactory(接口)工厂
SqlSessionFactoryBuilder ssfb = new SqlSessionFactoryBuilder();

SqlSessionFactory ssf = ssfb.build(in);
//3.使用工厂生产SqlSession对象
SqlSession ss = ssf.openSession();
//4.使用SqlSession创建dao接口的代理对象
StudentInte stu = ss.getMapper(StudentInte.class);
//插入信息

student.setName("哈哈");
student.setAge(23);
student.setScore(99.0);
stu.insertStudent(student);

//修改信息
student.setId(3);
student.setName("天天");
student.setAge(55);
student.setScore(45.4);
stu.updateStudent(student);

//删除信息
stu.deleteStudent(2);

//按id查询
List<User> listId =stu.findIdStudent(1);
System.out.println(listId);
System.out.println("---------------");

//查询所有
List<User> list = stu.findAll();

System.out.println(list);
ss.commit();//完成之后一定提交事务，否则数据库可能会刷新不出来

//关闭资源
in.close();
ss.close();

} catch (IOException e) {
e.printStackTrace();
}

}
}

第二种方式

不建立接口，直接使用xml开发方式，以用户表为例，属性有uname，usex，usex

第一步：建立user2用户表

第二步：在src目录下创建一个名为dao的包，在该包中创建持久化类MyUser，注意在类中声明的属性与数据表user2的字段一致

package dao;

public class MyUser {
private Integer uid;
private String uname;
private String usex;

public Integer getUid() {
return uid;
}

public void setUid(Integer uid) {
this.uid = uid;
}

public String getUname() {
return uname;
}

public void setUname(String uname) {
this.uname = uname;
}

public String getUsex() {
return usex;
}

public void setUsex(String usex) {
this.usex = usex;
}

@Override
public String toString() {
return "MyUser{" +
"uid=" + uid +
", uname='" + uname + '\'' +
", usex='" + usex + '\'' +
'}';
}
}

第三步：在resources文件夹下创建同名dao包，在其中创建MyUser的映射类UserMapper.xml

<?xml version="1.0" encoding="UTF-8" ?>
<!DOCTYPE mapper
PUBLIC "-//mybatis.org//DTD Mapper 3.0//EN"
"http://mybatis.org/dtd/mybatis-3-mapper.dtd">
<mapper namespace="dao/UserMapper.xml">
<!--namespace指代本文件的位置-->
<!--根据uid查询一个用户的信息-->
<select id="selectUserById" parameterType="Integer"
resultType="dao.MyUser">
select * from user2 where uid = #{uid}
</select>

<!--查询所有用户信息-->
<select id="selectAllUser" resultType="dao.MyUser">
select * from user2
</select>

<!--添加一个用户，#{uname}为dao.MyUser的属性值-->
<insert id="addUser" parameterType="dao.MyUser">
insert into user2 (uname,usex) value (#{uname},#{usex})
</insert>

<!--修改一个用户-->
<update id="updateUser" parameterType="dao.MyUser">
update user2 set uname = #{uname},usex = #{usex} where uid = #{uid}
</update>

<!--删除一个用户-->
<delete id="deleteUser" parameterType="Integer">
delete from user2 where uid = #{uid}
</delete>

</mapper>

第四步：直接在resources文件夹下创建核心配置文件mybatis-config.xml,在该文件中配置数据库环境和映射文件的位置

<?xml version="1.0" encoding="UTF-8" ?>
<!DOCTYPE configuration
PUBLIC "-//mybatis.org//DTD Config 3.0//EN"
"http://mybatis.org/dtd/mybatis-3-config.dtd">

<!--MyBatis的主配置文件-->
<configuration>
<!--配置环境-->
<environments default="mysql">
<!--配置mysql的环境，default取名无所谓，但指定之后id必须也为该指定-->
<environment id="mysql">
<!--配置事务的类型，当前事务为JDBC-->
<transactionManager type="JDBC"/>
<!--配置数据源（即数据库连接池）-->
<dataSource type="POOLED">
<!--配置数据库相关的四个基本信息-->
<property name="driver" value="com.mysql.jdbc.Driver"/>
<property name="url" value="jdbc:mysql://localhost:3306/test"/>
<property name="username" value="root"/>
<property name="password" value="1018"/>
</dataSource>
</environment>
</environments>

第五步：创建测试类MyBatisTest，在测试类中首先使用输入流读取配置文件，然后根据配置信息构建SqlSessionFactory对象。接下来通过SqlSessionFactory对象创建SQLSession对象，并使用SQLSession对象的方法进行数据库操作。

package mybatisTest;

​
import dao.MyUser;
import org.apache.ibatis.io.Resources;
import org.apache.ibatis.session.SqlSession;
import org.apache.ibatis.session.SqlSessionFactory;
import org.apache.ibatis.session.SqlSessionFactoryBuilder;

​
import java.io.IOException;
import java.io.InputStream;
import java.util.List;

public class Mybatis2_Test {
public static void main(String[] args) {
try {
//读取配置文件mybatis-config.xml
InputStream config = Resources.getResourceAsStream("mybatis-config.xml");

//根据配置文件构建SqlSessionFactory
SqlSessionFactory ssf = new SqlSessionFactoryBuilder().build(config);

//通过SqlSessionFactory创建SqlSession
SqlSession ss = ssf.openSession();

//SqlSession执行映射文件中的定义的SQL并返回映射
//dao/UserMapper.xml.selectUserById为UserMapper.xml中的命名空间+select的id
//查询一个用户
MyUser mu = ss.selectOne("dao/UserMapper.xml.selectUserById", 1);
System.out.println(mu);
System.out.println("-----------------");

//添加一个用户
MyUser addmu = new MyUser();
addmu.setUname("陈恒");
addmu.setUsex("男");
ss.insert("dao/UserMapper.xml.addUser", addmu);

//修改一个用户
MyUser updatemu = new MyUser();
updatemu.setUid(1);
updatemu.setUname("张三");
updatemu.setUsex("女");
ss.update("dao/UserMapper.xml.updateUser", updatemu);

//删除一个用户
ss.delete("dao/UserMapper.xml.deleteUser", 3);

//查询所有用户
List<MyUser> listMu = ss.selectList("dao/UserMapper.xml.selectAllUser");
for (MyUser myUser:listMu){
System.out.println(myUser);
}

//提交事务
ss.commit();

//关闭资源
ss.close();
config.close();
} catch (IOException e) {
e.printStackTrace();
}
}

}

ps：

1，接口类StudentInte所创方法名必须与UserMapper.xml映射文件中增删改查的 id一致，否则会报错。

2，事务完成后注意提交且关闭资源文件

3.注意及时配置约束文件

4.建立接口但是不建立实现类，使用SqlSession创建dao接口的代理对象

5.没有接口时，需要在测试类中用到一系列的select（s，o），inserrt（s，o），
delete（s，o），update（s，o）方法，具体用法视情况而定，前者s指代的映射文件中的id，o指的是参数。

6.namespace的用法：namespace不能为空，甚至可以用“xxx”进行代替，当有接口参与程序时，namespace指定为该接口的相对路径，id为该接口的指定方法名，起到了一一对应的作用，不至于吧众多接口方法搞混。当没有接口参与时，namespace为本映射文件的位置，s为“映射文件相对路径.id名”（ss.insert(“dao/UserMapper.xml.addUser”, addmu);），同样起到唯一标识的作用。

7.开发MyBatis程序时，一般不写实现类，那样会使代码累赘
